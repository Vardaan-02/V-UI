Loader

The BlockSwapLoader is a dynamic loader component where blocks periodically swap positions with smooth animations using Framer Motion. It simulates a lively, engaging visual effect by swapping two random blocks every 1.5 seconds.

Installation

1

Install Dependencies

npm i motion clsx tailwind-merge
2

Install Dependencies For Animations

npm install framer-motion
3

Add Util File

lib/utils.ts

import { ClassValue, clsx } from "clsx";
import { twMerge } from "tailwind-merge";


export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

Copy The Source Code

components/ui/block-swap-loader.tsx

1"use client";
2

3import React, { useState, useEffect } from "react";
4import { motion } from "framer-motion";
5

6const BlockSwapLoader: React.FC = () => {
7  const [blocks, setBlocks] = useState([0, 1, 2, 3, 4]);
8  const [animatingBlocks, setAnimatingBlocks] = useState<number[]>([]);
9  const [animationKey, setAnimationKey] = useState(0);
10

11  useEffect(() => {
12    const swapBlocks = () => {
13      const index1 = Math.floor(Math.random() * 5);
14      let index2;
15      do {
16        index2 = Math.floor(Math.random() * 5);
17      } while (index2 === index1);
18

19      setAnimatingBlocks([index1, index2]);
20      setAnimationKey((prevKey) => prevKey + 1);
21      setBlocks((prevBlocks) => {
22        const newBlocks = [...prevBlocks];
23        const temp = newBlocks[index1];
24        newBlocks[index1] = newBlocks[index2];
25        newBlocks[index2] = temp;
26        return newBlocks;
27      });
28    };
29

30    const intervalId = setInterval(swapBlocks, 1500);
31

32    return () => clearInterval(intervalId);
33  }, []);
34

35  return (
36    <div className="flex items-center justify-center gap-1">
37      {blocks.map((block, index) => {
38        const first = animatingBlocks[0] === index;
39        const second = animatingBlocks[1] === index;
40

41        const diff = animatingBlocks[1] - animatingBlocks[0];
42

43        return (
44          <motion.div
45            key={`${block}-${animationKey}`}
46            animate={first ? "up" : second ? "down" : false}
47            variants={{
48              up: {
49                y: [0, -36, -36, 0],
50                x: [0, 0, diff * 36, diff * 36],
51              },
52              down: {
53                y: [0, 36, 36, 0],
54                x: [
55                  0,
56                  0,
57                  diff * -36,
58                  diff * -36,
59                ],
60              },
61            }}
62            transition={{
63              duration: 1,
64              repeat: 0,
65            }}
66            className="w-8 h-8 bg-primary rounded-sm"
67          />
68        );
69      })}
70    </div>
71  );
72};
73

74export default BlockSwapLoader;
