Range Picker

This RangeSlider component allows users to select a range between two values (min and max) with draggable handles, dynamically updating the selected range and displaying the active portion visually. It uses state management, event listeners, and percentage calculations to handle interactions and style the slider accordingly.

1"use client";
2

3import type React from "react";
4import { useState, useCallback, useRef, useEffect } from "react";
5

6interface RangeSliderProps {
7  min: number;
8  max: number;
9  step: number;
10  initialValue1: number;
11  initialValue2: number;
12}
13

14export default function RangeSlider({
15  min,
16  max,
17  step,
18  initialValue1,
19  initialValue2,
20}: RangeSliderProps) {
21  const [value1, setValue1] = useState(initialValue1);
22  const [value2, setValue2] = useState(initialValue2);
23  const [isDragging, setIsDragging] = useState<"min" | "max" | null>(null);
24  const sliderRef = useRef<HTMLDivElement>(null);
25

26  const getPercentage = useCallback(
27    (value: number) => ((value - min) / (max - min)) * 100,
28    [min, max]
29  );
30

31  const handleMouseDown = useCallback(
32    (event: React.MouseEvent, handle: "min" | "max") => {
33      event.preventDefault();
34      setIsDragging(handle);
35    },
36    []
37  );
38

39  const handleMouseUp = useCallback(() => {
40    setIsDragging(null);
41  }, []);
42

43  const handleMouseMove = useCallback(
44    (event: MouseEvent) => {
45      if (!isDragging || !sliderRef.current) return;
46

47      const sliderRect = sliderRef.current.getBoundingClientRect();
48      const newPercentage =
49        ((event.clientX - sliderRect.left) / sliderRect.width) * 100;
50      const newValue = Math.round((newPercentage / 100) * (max - min) + min);
51

52      if (isDragging === "min") {
53        setValue1(Math.min(Math.max(newValue, min), value2 - step));
54      } else {
55        setValue2(Math.max(Math.min(newValue, max), value1 + step));
56      }
57    },
58    [isDragging, min, max, step, value1, value2]
59  );
60

61  useEffect(() => {
62    if (isDragging) {
63      window.addEventListener("mousemove", handleMouseMove);
64      window.addEventListener("mouseup", handleMouseUp);
65    }
66

67    return () => {
68      window.removeEventListener("mousemove", handleMouseMove);
69      window.removeEventListener("mouseup", handleMouseUp);
70    };
71  }, [isDragging, handleMouseMove, handleMouseUp]);
72

73  return (
74    <div className="w-full max-w-md mx-auto p-6">
75      <div className="relative w-full h-6 mb-4" ref={sliderRef}>
76        <div
77          className="absolute h-2 bg-gray-300 rounded-full top-2"
78          style={{
79            left: `0%`,
80            width: `${getPercentage(value1)}%`,
81          }}
82        />
83        <div
84          className="absolute h-2 bg-primary rounded-full top-2"
85          style={{
86            left: `${getPercentage(value1)}%`,
87            width: `${getPercentage(value2) - getPercentage(value1)}%`,
88          }}
89        />
90        <div
91          className="absolute h-2 bg-gray-300 rounded-full top-2"
92          style={{
93            left: `${getPercentage(value2)}%`,
94            width: `${100 - getPercentage(value2)}%`,
95          }}
96        />
97        <div
98          className="absolute w-6 h-6 bg-primary rounded-full top-0 -ml-3 cursor-pointer"
99          style={{ left: `${getPercentage(value1)}%` }}
100          onMouseDown={(e) => handleMouseDown(e, "min")}
101        />
102        <div
103          className="absolute w-6 h-6 bg-primary rounded-full top-0 -ml-3 cursor-pointer"
104          style={{ left: `${getPercentage(value2)}%` }}
105          onMouseDown={(e) => handleMouseDown(e, "max")}
106        />
107      </div>
108      <div className="flex justify-between mb-4">
109        <span className="text-sm font-medium text-gray-700">{value1}</span>
110        <span className="text-sm font-medium text-gray-700">{value2}</span>
111      </div>
112      <div className="text-center">
113        <p className="text-lg font-semibold text-gray-800">
114          Output: {value1} - {value2}
115        </p>
116      </div>
117    </div>
118  );
119}
