Toaster

The Toaster component is a customizable toast notification system that displays messages with a title, description, and optional undo button. It supports timed auto-dismissal, dynamic positioning, and progress indication, controlled via a ref for programmatic triggering.

    npx shadcn@latest add button
    npx shadcn@latest add toast

    1"use client";
    2

    3import { Button } from "@/components/ui/button";
    4import { CircleCheck, X } from "lucide-react";
    5import {
    6  forwardRef,
    7  useCallback,
    8  useImperativeHandle,
    9  useMemo,
    10  useState,
    11} from "react";
    12import {
    13  Toast,
    14  ToastAction,
    15  ToastClose,
    16  ToastDescription,
    17  ToastProvider,
    18  ToastTitle,
    19  ToastViewport,
    20} from "./toast";
    21import { useProgressTimer } from "@/hooks/use-progress-timer";
    22

    23interface MasteredToastProps {
    24  undo?: boolean;
    25  position?:
    26    | "top-right"
    27    | "top-left"
    28    | "bottom-right"
    29    | "bottom-left"
    30    | "top"
    31    | "bottom"
    32    | null
    33    | undefined;
    34}
    35

    36export interface MasteredToastRef {
    37  showToast: (title: string, description: string) => void;
    38}
    39

    40const MasteredToast = forwardRef<MasteredToastRef, MasteredToastProps>(
    41  ({ undo, position }, ref) => {
    42    const [open, setOpen] = useState(false);
    43    const [title, setTitle] = useState("");
    44    const [description, setDescription] = useState("");
    45    const toastDuration = 5000;
    46

    47    const { progress, start, pause, resume, reset } = useProgressTimer({
    48      duration: toastDuration,
    49      onComplete: () => setOpen(false),
    50    });
    51

    52    const convert = useMemo(() => {
    53      if (!position) return;
    54      else if (position.includes("right")) return "right";
    55      else if (position.includes("left")) return "left";
    56      else if (position.includes("top")) return "top";
    57      else return "bottom";
    58    }, [position]);
    59

    60    const handleOpenChange = useCallback(
    61      (isOpen: boolean) => {
    62        setOpen(isOpen);
    63        if (isOpen) {
    64          reset();
    65          start();
    66        }
    67      },
    68      [reset, start]
    69    );
    70

    71    const showToast = useCallback(
    72      (toastTitle: string, toastDescription: string) => {
    73        setTitle(toastTitle);
    74        setDescription(toastDescription);
    75        if (open) {
    76          setOpen(false);
    77          window.setTimeout(() => {
    78            handleOpenChange(true);
    79          }, 150);
    80        } else {
    81          handleOpenChange(true);
    82        }
    83      },
    84      [open, handleOpenChange]
    85    );
    86

    87    useImperativeHandle(ref, () => ({
    88      showToast,
    89    }));
    90

    91    return (
    92      <ToastProvider>
    93        <Toast
    94          open={open}
    95          onOpenChange={handleOpenChange}
    96          onMouseEnter={pause}
    97          onMouseLeave={resume}
    98          position={convert}
    99        >
    100          <div className="flex w-full justify-between gap-3">
    101            <CircleCheck
    102              className="mt-0.5 shrink-0 text-emerald-500"
    103              size={16}
    104              strokeWidth={2}
    105              aria-hidden="true"
    106            />
    107            <div className="flex grow flex-col gap-3">
    108              <div className="space-y-1">
    109                <ToastTitle>{title}</ToastTitle>
    110                <ToastDescription>{description}</ToastDescription>
    111              </div>
    112              {undo && (
    113                <div>
    114                  <ToastAction altText="Undo changes" asChild>
    115                    <Button size="sm" className="text-primary font-semibold">
    116                      Undo changes
    117                    </Button>
    118                  </ToastAction>
    119                </div>
    120              )}
    121            </div>
    122            <ToastClose asChild>
    123              <Button
    124                variant="ghost"
    125                className="group -my-1.5 -me-2 size-8 shrink-0 p-0 hover:bg-transparent"
    126                aria-label="Close notification"
    127              >
    128                <X
    129                  size={16}
    130                  strokeWidth={2}
    131                  className="opacity-60 transition-opacity group-hover:opacity-100"
    132                  aria-hidden="true"
    133                />
    134              </Button>
    135            </ToastClose>
    136          </div>
    137          <div className="contents" aria-hidden="true">
    138            <div
    139              className="pointer-events-none absolute bottom-0 left-0 h-1 w-full bg-emerald-500"
    140              style={{
    141                width: `${(progress / toastDuration) * 100}%`,
    142                transition: "width 100ms linear",
    143              }}
    144            />
    145          </div>
    146        </Toast>
    147        <ToastViewport position={position} />
    148      </ToastProvider>
    149    );
    150  }
    151);
    152

    153MasteredToast.displayName = "MasteredToast";
    154

    155export default MasteredToast;

    components/ui/use-toast.tsx

    1import { useRef, useCallback } from "react";
    2import MasteredToast, {
    3  MasteredToastRef,
    4} from "@/components/ui/toast-mastered";
    5

    6interface UseToastOptions {
    7  undo?: boolean;
    8  position?:
    9    | "top-right"
    10    | "top-left"
    11    | "bottom-right"
    12    | "bottom-left"
    13    | "top"
    14    | "bottom"
    15    | null
    16    | undefined;
    17}
    18

    19export function useToast(options: UseToastOptions) {
    20  const toastRef = useRef<MasteredToastRef | null>(null);
    21

    22  const showToast = useCallback(
    23    (title: string, description: string) => {
    24      toastRef.current?.showToast(title, description);
    25    },
    26    []
    27  );
    28  
    29  const Toaster = (
    30    <MasteredToast
    31      ref={toastRef as React.RefObject<MasteredToastRef>}
    32      undo={options.undo}
    33      position={options.position}
    34    />
    35  );
    36

    37  return { showToast, Toaster };
    38}
